MySolution.sln
 ├─ src/
 │   ├─ MyApp.Api/             # ASP.NET Core Web API (presentation)
 │   │    ├─ Controllers/
 │   │    ├─ Middlewares/      # e.g. error handling
 │   │    ├─ Models/           # API-specific DTOs or ViewModels
 │   │    └─ Program.cs
 │   ├─ MyApp.Application/     # Application layer (use cases, services)
 │   │    ├─ Features/         # (e.g. Orders, Customers, each with Commands/Queries)
 │   │    ├─ DTOs/             # Data Transfer Objects
 │   │    └─ Interfaces/       # e.g. IDataRepository, IEmailSender 
 │   ├─ MyApp.Domain/          # Domain layer (core business)
 │   │    ├─ Entities/         # Domain entities (e.g. Order.cs, Customer.cs)
 │   │    ├─ ValueObjects/     # e.g. Address.cs, Money.cs
 │   │    ├─ Events/           # Domain events
 │   │    ├─ Exceptions/       # Domain-specific exceptions
 │   │    └─ Interfaces/       # Repository/Domain service interfaces
 │   └─ MyApp.Infrastructure/  # Infrastructure (EF Core, external services)
 │        ├─ Persistence/
 │        │    ├─ EntityConfigurations/  # EF Fluent API configs
 │        │    ├─ Migrations/            # EF migrations
 │        │    ├─ Repositories/          # EF repository implementations
 │        │    └─ MyAppDbContext.cs
 │        └─ Services/                    # e.g. EmailService, PaymentGatewayClient
 └─ tests/
     ├─ MyApp.UnitTests/       # Unit tests (xUnit/NUnit/MSTest)
     └─ MyApp.IntegrationTests/# Integration tests (real DB, external services)
